//     Модифицируйте права доступа и владельца файла index.html следующим образом: Текущий пользователь должен иметь права на чтение/запись Группа www-data на чтение Все остальные не имеют доступа к файлу
# 4. Модифицируем права доступа и владельца файла index.html следующим образом
    # Меняем группу пользователей файла index.html (sudo изменения от имени пользователя root ( chgrp - change group изменить группу ) (имя группы www-data) (index.html - файл к которому применяются изменения))
        sudo chgrp www-data index.thml
    # Изменение прав доступа к файлу index.html текущего пользователя
        chmod u-rwx index.html
        chmod u+rw index.html 
    # Права доступа группы www-data устанавливаем только чтение
        chmod g-rwx index.html
        chmod g+r index.html
    # Правадоступа остальных пользователей устанавливаем
        chmod o-rwx index.html
