//     На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403
# 3.  На основании файла 404.html создаём файл 403.html заменив в его содержимом все упоминания 404 на 403
    # Переходим из папки logs в папку html
        cd ../www/html/
    # создаем копию файла 404.html с именем 403.html
        cp 404.html 403.html
    # меняем в файле все упоминания 404 на 403 (-i сохраняем в исходный файл 403.html ( s команда замены) ( 404/403 менгяем 404 на 403 ) ( g меняем глобально, т. е. все, а  не только первое вхождение ) (403.html имя файла в котором производим замену))
        sed -i 's/404/403/g' 403.html
